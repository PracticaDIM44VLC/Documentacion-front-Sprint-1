import { Component, OnInit } from '@angular/core';
import { Location } from '@angular/common';
import { Operation } from 'src/app/models/operation';
import { OperationImpl } from 'src/app/models/operation-impl';
import { DataService } from 'src/app/service/data.service';

@Component({
  selector: 'app-layouts',
  templateUrl: './layouts.component.html',
  styleUrls: ['./layouts.component.css']
})
export class LayoutsComponent implements OnInit {
  public operation: Operation = new OperationImpl();
  operationData: Operation = new OperationImpl();
  operations: Operation[] = [];

  constructor(private _location: Location, private dataService: DataService) { }

  ngOnInit(): void {
    this.dataService.getOperations().subscribe((response) =>
      this.operations = this.dataService.extraerOperations(response));
  }

  verDatos(operation: Operation): void {
    this.operationData = operation;
  }

  goBack(): void {
    this._location.back();
  }

}
