import { Component, OnInit } from '@angular/core';
import { Location } from '@angular/common';
import { ActivatedRoute } from '@angular/router';
import { faLaptop, faPhone, faPrint, faBarcode, faVideo } from '@fortawesome/free-solid-svg-icons';
import { Branch } from 'src/app/models/branch';
import { BranchImpl } from 'src/app/models/branch-impl';
import { Device } from 'src/app/models/device';
import { DutyTitle } from 'src/app/models/duty-title';
import { Location as LocationClass } from 'src/app/models/location';
import { LocationImpl } from 'src/app/models/location-impl';
import { DataService } from 'src/app/service/data.service';

@Component({
  selector: 'app-layout-branch',
  templateUrl: './layout-branch.component.html',
  styleUrls: ['./layout-branch.component.css']
})
export class LayoutBranchComponent implements OnInit {
  id: string = "";

  branch: Branch = new BranchImpl();
  location: LocationClass = new LocationImpl();

  locations: LocationClass[] = [];
  dutytitles: DutyTitle[] = [];
  devices: Device[] = [];

  // ICONS
  laptop = faLaptop;
  phone = faPhone;
  printer = faPrint;
  scanner = faBarcode;
  vtc = faVideo;

  constructor(private _location: Location, private activateRoute: ActivatedRoute, private dataService: DataService) { }

  ngOnInit(): void {
    this.id = this.activateRoute.snapshot.params['id'];

    // Mapear title Branch (Leyenda)
    this.dataService.getBranch(this.id).subscribe((response) =>
      this.branch = this.dataService.mapearBranch(response));

    this.dataService.getLocationsBranch(this.id).subscribe((response) => {
      this.locations = this.dataService.extraerLocation(response);
    });

    this.dataService.getDutyTitlesBranch(this.id).subscribe((response) => {
      this.dutytitles = this.dataService.extraerDutyTitle(response);
    });

    this.dataService.superGetDevices(this.id).subscribe((response) => {
      this.devices = this.dataService.extraerDevice(response);
    });
  }

  goBack(): void {
    this._location.back();
  }

}
