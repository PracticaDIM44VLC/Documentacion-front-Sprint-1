import { Component, OnInit } from '@angular/core';
import { Location } from '@angular/common';
import { ActivatedRoute } from '@angular/router';
import { Device } from 'src/app/models/device';
import { DutyTitle } from 'src/app/models/duty-title';
import { DutyTitleImpl } from 'src/app/models/duty-title-impl';
import { Location as LocationClass } from 'src/app/models/location';
import { DataService } from 'src/app/service/data.service';
import { environment } from 'src/environments/environment';
import { Branch } from 'src/app/models/branch';
import { BranchImpl } from 'src/app/models/branch-impl';

@Component({
  selector: 'app-admin-dutytitle-form',
  templateUrl: './admin-dutytitle-form.component.html',
  styleUrls: ['./admin-dutytitle-form.component.css']
})
export class AdminDutytitleFormComponent implements OnInit {
  id: string = "";
  host: string = environment.host;

  public dutytitle: DutyTitle = new DutyTitleImpl();
  dutytitleData: DutyTitle = new DutyTitleImpl();
  dutytitles: DutyTitle[] = [];

  branch: Branch = new BranchImpl();

  locations: LocationClass[] = [];

  constructor(private _location: Location, private activateRoute: ActivatedRoute, private dataService: DataService) {}

  ngOnInit(): void {
    this.id = this.activateRoute.snapshot.params['id'];
    this.dataService.getDutyTitle(this.id).subscribe((response) => {
      this.dutytitle = this.dataService.mapearDutyTitle(response);
      this.dataService.superGetLocation(this.dutytitle.idBranchOfThisDutyTitle).subscribe((response) => {
        this.locations = this.dataService.extraerLocation(response);
      });
    });

    setTimeout(() => {
      this.dataService.getBranchUrl(this.dutytitle.branch).subscribe((response) =>
        this.branch = this.dataService.mapearBranch(response));
    }, 1000);

  }

  verDatos(dutytitle: DutyTitle): void {
    this.dutytitleData = dutytitle;
  }

  assign(dutytitle: DutyTitle): void {
    this.dataService.updateDT(dutytitle).subscribe();
  }

  addDT(): void {
    this.dutytitle.branch = `${this.host}branches/${this.id}`;
    this.dataService.createDT(this.dutytitle).subscribe((response) =>
    this.ngOnInit());
  }

  goBack(): void {
    this._location.back();
  }
}
