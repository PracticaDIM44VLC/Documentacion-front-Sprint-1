import { Component, Input, OnInit } from '@angular/core';
import { Router, ActivatedRoute } from '@angular/router';
import { AreaCP } from 'src/app/models/area-cp';
import { Branch } from 'src/app/models/branch';
import { CommandPost } from 'src/app/models/command-post';
import { Location } from 'src/app/models/location';
import { LocationImpl } from 'src/app/models/location-impl';
import { Operation } from 'src/app/models/operation';
import { OperationImpl } from 'src/app/models/operation-impl';
import { TypeLocation } from 'src/app/models/type-location';
import { DataService } from 'src/app/service/data.service';
import { environment } from 'src/environments/environment';

@Component({
  selector: 'app-superadmin-location',
  templateUrl: './superadmin-location.component.html',
  styleUrls: ['./superadmin-location.component.css']
})
export class SuperadminLocationComponent implements OnInit {
  id: string = "";
  host: string = environment.host;

  @Input() operation: Operation = new OperationImpl();
  operationData: Operation = new OperationImpl();

  public location: Location = new LocationImpl();
  locationData: Location = new LocationImpl();
  locations: Location[] = [];

  branches: Branch[] = [];

  keys = Object.keys;
  types = TypeLocation;
  areas = AreaCP;
  cps = CommandPost;

  constructor(private router: Router, private activateRoute: ActivatedRoute, private dataService: DataService ) { }

  ngOnInit(): void {
    this.id = this.activateRoute.snapshot.params['id'];
    this.dataService.getLocationsOperation(this.id).subscribe((response) =>
      this.locations = this.dataService.extraerLocation(response));
    this.dataService.superGetBranch(this.id).subscribe((response) => {
      this.branches = this.dataService.extraerBranch(response)});
    this.dataService.getOperation(this.id).subscribe((response) =>
      this.operation = this.dataService.mapearOperations(response));
  }

  verDatos(location: Location): void {
    this.locationData = location;
  }

  addLoc(): void {
    this.location.operation = `${this.host}operations/${this.id}`;
    this.dataService.createLoc(this.location).subscribe((response) =>
    this.ngOnInit());
  }

  confirm(): void {
    this.router.navigate(['/superadmin']);
  }

}
