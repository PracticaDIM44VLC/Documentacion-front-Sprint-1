import { HttpClient } from '@angular/common/http';
import { Injectable } from '@angular/core';

  // mtodo para conseguir el ID
  getId(url: string): string {
    let posicionFinal: number = url.lastIndexOf('/');
    let numId: string = url.slice(posicionFinal + 1, url.length);

    return numId;
  }

  /*
   * Mapear datos Operation
   */
  extraerOperation(respuestaApi: any): Operation[] {
    const operations: Operation[] = [];
    let respuesta: any = respuestaApi._embedded.operations;
    if (respuesta === undefined) {
      console.info("No Operation");
    } else
    respuestaApi._embedded.operations.forEach((op: any) => {
      operations.push(this.mapearOperation(op));
    });

    return operations;
  }
  mapearOperation(operationApi: any): Operation {
    let operation: Operation = new OperationImpl();
    operation.idOperation = this.getId(operationApi._links.operation.href);
    operation.nameOperation = operationApi.nameOperation;
    operation.unitsHref = operationApi._links.unitsOnThisOperation.href;
    operation.locationsHref = operationApi._links.locationsInThisOperation.href;

    return operation;
  }

  extraerOperations(respuestaApi: any): Operation[] {
    const operations: Operation[] = [];
    let respuesta: any = respuestaApi._embedded.operations;
    if (respuesta === undefined) {
      console.info("No Operations");
    } else
    respuestaApi._embedded.operations.forEach((op: any) => {

    operation.idOperation = this.getId(operationApi._links.operation.href);

  }

    let respuesta: any = respuestaApi._embedded.locations;
    if (respuesta === undefined) {
      console.info("No Locations");
    } else
    respuestaApi._embedded.locations.forEach((loc: any) => {
      locations.push(this.mapearLocation(loc));
    });

    return locations;
  }

  mapearLocation(locationApi: any): Location {
    let location: Location = new LocationImpl();
    location.idLocation = this.getId(locationApi._links.location.href);
    location.nameLocation = locationApi.nameLocation;
    location.typeLocation = locationApi.typeLocation;
    // location.areaCommandPost = locationApi.areaCommandPost;
    // location.commandPost = locationApi.commandPost;
    location.branch = locationApi._links.branch.href;
    location.operation = locationApi._links.operation.href;
    location.dutyTitlesInThisLocation = locationApi._links.dutyTitlesInThisLocation.href;
    location.url = locationApi._links.self.href;

    return location;
  }
